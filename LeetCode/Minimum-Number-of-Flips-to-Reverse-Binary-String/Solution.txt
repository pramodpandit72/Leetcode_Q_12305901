class Solution {
public:
    int minimumFlips(int n) {
        string s;
        while (n>0){
            s+=(n%2)+'0';
            n/=2;
        }
        reverse(s.begin(), s.end());

        int f=0;
        int L=s.size();

        for(int i=0;i<L/2;i++){
            if(s[i] != s[L-1-i])
                f++;
        }
        return 2*f;
    }
};
